<script>
        $(document).ready(function () {
            $("#form_name").keyup(function () {
                var value = $("#form_name").val();
                if (value.length == 0) {
                    $("#menuFull figure").show();
                }
                else {
                    $("#menuFull figure").hide();
                    $("#menuFull figure:contains(" + value + ")").addClass("testt").show();  
                }
            })
        })
    </script>
